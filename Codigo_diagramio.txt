Table Planos {
  plano_id serial [not null, pk, increment]
  nome varchar(40) [not null]
  meses int [not null]        
  preco_mensal decimal(8,2) [not null] 
  descricao varchar(200)
  ativo boolean [default: true]
}

Table Instrutores {
  instrutor_id serial [not null, pk, increment]
  nome varchar(80) [not null]
  cpf char(11) [unique]
  telefone varchar(20)
  registro_CREF varchar(10) [default: 'Pleno']
  ativo boolean [default: true]
  contratado_em date
}

Table Modalidades {
  modalidade_id serial [not null, pk, increment]
  nome varchar(60) [not null]
  descricao varchar(200)
  instrutor_principal_id int [ref: > Instrutores.instrutor_id]
  ativo boolean [default: true]
}

Table Treinos {
  treino_id serial [not null, pk, increment]
  modalidade_id int [not null, ref: > Modalidades.modalidade_id]
  instrutor_id int [not null, ref: > Instrutores.instrutor_id]
  data_hora datetime [not null]
  duracao_min tinyint [not null]  
  local varchar(80) [not null]
  max_vagas smallint [default: 30]
}

Table Alunos {
  aluno_id serial [not null, pk, increment]
  nome varchar(100) [not null]
  cpf char(11) [unique]
  email varchar(100) [unique]
  telefone varchar(20)
  data_nascimento date
  sexo varchar(1) [default: 'O'] 
  criado_em datetime [default: 'CURRENT_TIMESTAMP']
  ativo boolean [default: true]
}

Table Assinatura {
  assinatura_id serial [not null, pk, increment]
  aluno_id int [not null, ref: > Alunos.aluno_id]
  plano_id int [not null, ref: > Planos.plano_id]
  inicio date [not null]
  termino date [not null]          
  status boolean [default: true]
  criado_em datetime [default: 'CURRENT_TIMESTAMP']
}

Table Pagamento {
  pagamento_id serial [not null, pk, increment]
  assinatura_id int [not null, ref: > Assinatura.assinatura_id]
  valor decimal(8,2) [not null]   
  data_pagamento date [not null]
  metodo varchar(10) [default: 'Pix']
  status boolean [default: true]
  referencia varchar(80)
}

Table Presencas {
  presenca_id serial [not null, pk, increment]
  aluno_id int [not null, ref: > Alunos.aluno_id]
  treino_id int [not null, ref: > Treinos.treino_id]
  presente boolean [default: false]
  registrado_em datetime [default: 'CURRENT_TIMESTAMP']
  observacao varchar(200)

}
